void IndexCSC(par_t *par, mat_t *mat) {

 int k;
 if (((mat->colind) && (mat->colval) && (mat->rowptr)) == 0) {
  fprintf(stderr, "The CSR form is not ready yet");
 }
 for (k = 0; k < mat->nnz; ++k) {
  (mat->marked)[mat->colind[k]]++;
 }

 mat->bsize = 0;
 int currentblk = 0;
 for (k = 0; k < mat->ncol; ++k) {

  if (k == 0)
   mat->colptr[k] = 0;
  else
   mat->colptr[k] = mat->colptr[k - 1] + mat->marked[k - 1];

  if (mat->marked[k] != 0) {
   mat->bsize++;
   while (k >= par->blkrowst[currentblk + 1]) {
    currentblk++;
   }
  }
 }
 mat->colptr[k] = mat->nnz;
 int *colacc;
 do{ colacc=(int*)calloc((mat->ncol),sizeof(int)); if ((colacc)==((void *)0)) fprintf(stderr,"Malloc failure PTR=%s @ Line %d\t %s\n","colacc",524,"colptr accummulator"); }while(0);

 int r, cpt, c, nnzcounter = 0;
 for (r = 0; r < mat->nrow; ++r) {
  for (cpt = mat->rowptr[r]; cpt < mat->rowptr[r + 1]; ++cpt) {
   nnzcounter++;
   c = mat->colind[cpt];
   mat->rowval[mat->colptr[c] + colacc[c]] = mat->colval[cpt];
   mat->rowind[mat->colptr[c] + colacc[c]] = r;
   colacc[c]++;
  }
 }
 do{ if (colacc==((void *)0)) fprintf(stderr,"Empty pointer freed PTR=%s @ Line %d\t %s\n","colacc",536,"accumulator for each column");else{ free((void*) colacc);} }while(0);
}

void DeterminNonlocalb(par_t *par, mat_t *mat) {
 int k;

 do{ mat->nonlocalbst=(int*)calloc((par->np + 1),sizeof(int)); if ((mat->nonlocalbst)==((void *)0)) fprintf(stderr,"Malloc failure PTR=%s @ Line %d\t %s\n","mat->nonlocalbst",542,"Mat blkbsize"); }while(0);

 do{ mat->needbidx=(int*)malloc(sizeof(int)*(mat->bsize)); if ((mat->needbidx)==((void *)0)) fprintf(stderr,"Malloc failure PTR=%s @ Line %d\t %s\n","mat->needbidx",544,""); }while(0);

 int nowptrb = 0;
 int blkcounter = 0;

 for (k = 0; k < mat->ncol; ++k) {

  if (k == par->blkrowst[blkcounter]) {
   mat->nonlocalbst[blkcounter] = nowptrb;
   ++blkcounter;
  }

  if (mat->marked[k] != 0) {
   mat->needbidx[nowptrb] = k;
   ++nowptrb;
  }
 }
 mat->nonlocalbst[blkcounter] = mat->bsize;
}

void TransferVector(par_t *par, mat_t *mat) {
 int k;
 MPI_Status status;

 do{ mat->b=(double*)malloc(sizeof(double)*(mat->bsize)); if ((mat->b)==((void *)0)) fprintf(stderr,"Malloc failure PTR=%s @ Line %d\t %s\n","mat->b",568," alloc space for all needed b"); }while(0);


 for (k = mat->nonlocalbst[par->id]; k < mat->nonlocalbst[par->id + 1];
   ++k) {
  mat->b[k] = par->localvec[mat->needbidx[k] - par->blkrowst[par->id]];
 }


 int idstep;
 int outdst, insrc;
 int sendbufsize = 0, recvbufsize = 0;
 int *recvidxbuf = ((void *)0);
 double *sendvalbuf = ((void *)0);


 for (idstep = 1; idstep < par->np; ++idstep) {
  outdst = ((par->id) + idstep) % (par->np);
  insrc = ((par->id) + par->np - idstep) % (par->np);


  sendbufsize = mat->nonlocalbst[outdst + 1] - mat->nonlocalbst[outdst];
  MPI_Send(&sendbufsize, 1, (&ompi_mpi_int), outdst, 11, (&ompi_mpi_comm_world));





  MPI_Recv(&recvbufsize, 1, (&ompi_mpi_int), insrc, 11, (&ompi_mpi_comm_world), &status);

  if (recvbufsize != 0) {
   do{ recvidxbuf=(int*)malloc(sizeof(int)*(recvbufsize)); if ((recvidxbuf)==((void *)0)) fprintf(stderr,"Malloc failure PTR=%s @ Line %d\t %s\n","recvidxbuf",
 600
# 599 "matvec_mpi.c"
   ,"recvbuf get requested index"); }while(0)
                                   ;
   do{ sendvalbuf=(double*)malloc(sizeof(double)*(recvbufsize)); if ((sendvalbuf)==((void *)0)) fprintf(stderr,"Malloc failure PTR=%s @ Line %d\t %s\n","sendvalbuf",
 602
# 601 "matvec_mpi.c"
   ,"send out requested values"); }while(0)
                                 ;
  }


  if (sendbufsize != 0) {

   MPI_Bsend(&mat->needbidx[mat->nonlocalbst[outdst]], sendbufsize,
     (&ompi_mpi_int), outdst, 12, (&ompi_mpi_comm_world));


  }


  if (recvbufsize != 0)
   MPI_Recv(recvidxbuf, recvbufsize, (&ompi_mpi_int), insrc, 12,
     (&ompi_mpi_comm_world), &status);




  for (k = 0; k < recvbufsize; ++k) {
   sendvalbuf[k] =
     par->localvec[recvidxbuf[k] - par->blkrowst[par->id]];
  }


  MPI_Bsend(sendvalbuf, recvbufsize, (&ompi_mpi_double), insrc, 13,
    (&ompi_mpi_comm_world));




  MPI_Recv(&mat->b[mat->nonlocalbst[outdst]], sendbufsize, (&ompi_mpi_double),
    outdst, 13, (&ompi_mpi_comm_world), &status);


  if (recvbufsize != 0) {
   do{ if (recvidxbuf==((void *)0)) fprintf(stderr,"Empty pointer freed PTR=%s @ Line %d\t %s\n","recvidxbuf",639,"recv index buffer freed");else{ free((void*) recvidxbuf);} }while(0);
   do{ if (sendvalbuf==((void *)0)) fprintf(stderr,"Empty pointer freed PTR=%s @ Line %d\t %s\n","sendvalbuf",640,"recv index buffer freed");else{ free((void*) sendvalbuf);} }while(0);
  }
 }
}

void ComputeOutput(par_t *par, mat_t *mat) {
 int k;

 do{ mat->vout=(double*)calloc((mat->nrow),sizeof(double)); if ((mat->vout)==((void *)0)) fprintf(stderr,"Malloc failure PTR=%s @ Line %d\t %s\n","mat->vout",648,"block output"); }while(0);
 int r, c;
 for (k = 0; k < mat->bsize; ++k) {
  c = mat->needbidx[k];
  for (r = mat->colptr[c]; r < mat->colptr[c + 1]; ++r) {
   mat->vout[mat->rowind[r]] += mat->b[k] * mat->rowval[r];
  }
 }
}

void GatherOutput(par_t *par, mat_t *mat) {
 int k;
 MPI_Status status;
 if (par->id == 0) {
  do{ par->finalout=(double*)malloc(sizeof(double)*(par->nrow)); if ((par->finalout)==((void *)0)) fprintf(stderr,"Malloc failure PTR=%s @ Line %d\t %s\n","par->finalout",662," final out"); }while(0);
  memcpy(par->finalout, mat->vout,
    sizeof(double) * (par->blkrowst[1] - par->blkrowst[0]));
  for (k = 1; k < par->np; ++k) {
   MPI_Recv(&par->finalout[par->blkrowst[k]],
     (par->blkrowst[k + 1] - par->blkrowst[k]), (&ompi_mpi_double), k,
     -1, (&ompi_mpi_comm_world), &status);
  }
 } else {

  MPI_Send(mat->vout, mat->nrow, (&ompi_mpi_double), 0, 0, (&ompi_mpi_comm_world));
 }
}

void WriteOutput(par_t *par, mat_t *mat) {
 int k;
 FILE *f;
 printf("Output file name: %s", par->fout);
 if (par->finalout == ((void *)0) ) {
  fprintf(stderr, "Final out vector is NULL\n");
 }
 if ((f = fopen(par->fout, "w")) == ((void *)0) ) {
  fprintf(stderr, "Fail to open output file: %s", par->fout);
  return;
 }
 (par->t12 -= MPI_Wtime());
 for (k = 0; k < par->nrow; ++k) {
  fprintf(f, "%.15lf\n", par->finalout[k]);
 }
 fclose(f);
 (par->t12 += MPI_Wtime());
}
